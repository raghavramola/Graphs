int primeMST(vector<int>graph[],int v)
{
int key[v],res=0;
memset(key,INT_MAX,sizeof(key));
key[0] = 0;
bool mset[v] = {false};
for(int count=0;count<v;count++)
{
int u = -1;
for(int i=0;i<v;i++)
{
if(!mset[i]&&(u==-1||key[i]<key[u])
u = i;
}
mset[u] = true;
res = res + key[u];
for(int j=0;j<v;i++)
{
if(graph[u][j]!=0&&!mset[j])
key[j] = min(key[j],graph[u][j]);
}
return res;
}
